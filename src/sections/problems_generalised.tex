    \problemlist{Generalised problems}
  \else

    \problemlist{Problems}

  \ifdefined\NOSOURCES

    \fi
  \else
    \newpage
  \fi

  \ifdefined\ORIGINAL
    \problemlist{Solutions to generalised problems}
  \else
    \problemlist{Solutions}
  \fi

  \ifdefined\ONLYSOLUTIONS
    \ifdefined\NOSOURCES
      \printsourcesfalse
    \else
      \printsourcestrue
    \fi
  \else
    \printsourcesfalse  % Avoid printing the sources again if they were already included in the statements-only part
  \fi
