\begin{equation*}
  \prod\limits_{\underset{\scriptstyle{p \text{ prime}}}{1 \leq p \leq n}} {\mspace{-8mu} p}  \mspace{2mu}
  \leq \mspace{2mu}
  4^{n - 1}
\end{equation*}
