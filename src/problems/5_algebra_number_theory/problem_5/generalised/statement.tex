\begin{enumerate}
 \item Do there exist rational numbers $a, b > 0$ such that $a^b$ is rational?
 \item Do there exist rational numbers $a, b > 0$ such that $a^b$ is irrational?
 \item Do there exist $a \in \mathbb{R}^{+}$ rational and $b \in \mathbb{R}^{+}$ irrational such that $a^b$ is rational?
 \item Do there exist $a \in \mathbb{R}^{+}$ rational and $b \in \mathbb{R}^{+}$ irrational such that $a^b$ is irrational?
 \item Do there exist $a \in \mathbb{R}^{+}$ irrational and $b \in \mathbb{R}^{+}$ rational such that $a^b$ is rational?
 \item Do there exist $a \in \mathbb{R}^{+}$ irrational and $b \in \mathbb{R}^{+}$ rational such that $a^b$ is irrational?
 \item Do there exist irrational numbers $a, b \in \mathbb{R}^{+}$ such that $a^b$ is rational?
 \item Do there exist irrational numbers $a, b \in \mathbb{R}^{+}$ such that $a^b$ is irrational?
\end{enumerate}
