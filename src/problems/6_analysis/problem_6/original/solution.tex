For each $N \in \mathbb{N}$, using partial fraction decomposition:
\begin{equation*}
\begin{split}
  \sum\limits_{n=1}^{N} {\, \frac{1}{n^3 + 3 n^2 + 2n}} \,
  &= \, \sum\limits_{n=1}^{N} {\, \frac{1}{n \paren{n + 1} \paren{n + 2}}} \,
   = \, \frac{1}{2} \, \sum\limits_{n=1}^{N} {\, \frac{1}{n} - \frac{2}{n + 1} + \frac{1}{n + 2}} \, = \\[1.35ex]
  &= \, \frac{1}{2} \, \sum\limits_{n=1}^{N} {\, \frac{1}{n}}
     \, + \, \frac{1}{2} \, \sum\limits_{n=3}^{N + 2} {\, \frac{1}{n}}
     \, - \, \sum\limits_{n=2}^{N + 1} {\, \frac{1}{n}} \, = \\[0.6ex]
  &= \, \frac{1}{2} \mspace{-1.2mu} + \mspace{-1.2mu} \frac{1}{4}
     \, + \, \frac{1}{2} \! \cdot \! \frac{1}{N + 1} \mspace{-0.6mu}
     + \mspace{-0.6mu} \frac{1}{2} \! \cdot \! \frac{1}{N + 2}
     \, - \mspace{1.5mu} \frac{1}{2} \mspace{-1.2mu} - \mspace{-1.2mu} \frac{1}{N + 1}
     \, + \, \sum\limits_{n=3}^{N} {\, \frac{1}{n} - \frac{1}{n}} \, = \\[0.3ex]
  &= \frac{1}{4}
     - \frac{1}{2} \! \cdot \! \frac{1}{N + 1}
     + \frac{1}{2} \! \cdot \! \frac{1}{N + 2}
     \, + \, 0 = \\[1.1ex]
  &= \frac{1}{4} - \frac{1}{2 \paren{N + 1} \paren{N + 2}} \,
   < \, \frac{1}{4}
\end{split}
\end{equation*}
